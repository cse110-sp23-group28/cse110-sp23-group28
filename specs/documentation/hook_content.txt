
# Run ESLint
eslint .

# Run Jest
jest

# Capture the exit status of both commands
ESLINT_STATUS=$?
JEST_STATUS=$?

# Check if either command failed
if [ $ESLINT_STATUS -ne 0 ] || [ $JEST_STATUS -ne 0 ]; then
  echo "Pre-commit failed. Please fix the linting and unit test errors before committing."
  exit 1
fi

# If all checks pass, allow the commit to proceed
exit 0
